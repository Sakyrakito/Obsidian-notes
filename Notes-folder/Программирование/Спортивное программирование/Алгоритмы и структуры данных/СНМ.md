struct DSU {
    int n;
    vector<int> p, r;
    int comps;
    DSU(int sz) {
        n = sz;
        p.resize(n);
        r.assign(n, 0);
        for (int i = 0; i < n; i++)
            p[i] = i;
        comps = n;
    }

    int find(int x) {
        return (p[x] == x ? x : p[x] = find(p[x]));
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b)
            return false;

        if (r[a] < r[b])
            swap(a, b);
        p[b] = a;

        if (r[a] == r[b])
            r[a]++;
        comps--;
        return true;
    }
};
