>дерево отрезков с обновлением на отрезке

C++:
```c++
class MassSegmentTree  {
public:
    MassSegmentTree(int sz) {
        n = sz;
        a.resize(sz);
        tree.resize(sz * 4);
        add.resize(sz * 4);
    }

    void update(int l, int r, int val) {
        update_tree(l, r, val, 0, 0, n - 1);
    }

    int get_num(int i) {
        return get_num_tree(i, 0, 0, n - 1);
    }

    void push(int v) {
        add[v * 2 + 1] += add[v];
        add[v * 2 + 2] += add[v];
        add[v] = 0;
    }

private:
    int n;
    vector<int> a, tree, add;

    void update_tree(int l, int r, int val, int v, int tl, int tr) {
        if (tl > r || tr < l)
            return;
        if (tl >= l && tr <= r) {
            add[v] += val;
            return;
        }
        push(v);
        int m = (tl + tr) / 2;
        update_tree(l, r, val, v * 2 + 1, tl, m);
        update_tree(l, r, val, v * 2 + 2, m + 1, tr);
    }


    int get_num_tree(int i, int v, int l, int r) {
        if (l == r) {
            tree[v] += add[v];
            add[v] = 0;
            return tree[v];
        }
        push(v);
        int m = (l + r) / 2;
        if (i <= m)
            return get_num_tree(i, v * 2 + 1, l, m);
        else
            return get_num_tree(i, v * 2 + 2, m + 1, r);
    }
};
```
